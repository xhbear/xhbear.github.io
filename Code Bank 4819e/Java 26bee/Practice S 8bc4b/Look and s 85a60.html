The program will accept an integer which corresponds to the number of iterations of the look-and-say algorithm and output the result.

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        PerformLookAndSayAlgo();
    }
    public static long[] ConvertLongToArrayOfDigits(long input) {
        int digits = CountDigitsInNumber(input);
        long[] arrOfDigits = new long[digits];
        for (int i = 0; i < digits; i++) {
            arrOfDigits[i] = input % 10;
            input = input / 10;
        }
        ReverseArrayElements(arrOfDigits);
        return arrOfDigits;
    } // 要記得和ReverseArrayElements函式搭配使用，把陣列裡面的單元倒回來排序
    public static void PerformLookAndSayAlgo() {
        Scanner sc = new Scanner(System.in);
        int numberOfCalculations = sc.nextInt();
        long algoResults = 0;
        long input = 11;
        long output = 0;
        for (int i = 0; i < numberOfCalculations - 2; i++) {
            output = LookSayAlgo(input);
            input = output;
        }
        algoResults = output;
        // initialize count = 1
        // count how many like numbers
        // String = String + save to String
        // counter++
        System.out.println(algoResults);
    }
    public static int CountDigitsInNumber (long input) {
        int count = 1;
        while (input >= 10) {
            input = input / 10;
            count++;
        }
        return count;
    }
    public static void ReverseArrayElements(long[] arr) {
        int length = arr.length;
        int count = length-1;
        long[] temp = new long[length];
        for (int i = 0; i < length; i++) {
            temp[i] = arr[count];
            count--;
        }
        for (int i = 0; i < length; i++) {
            arr[i] = temp[i];
        }
    }
    public static int CountSteps (long[] number) {
        int steps = 1;
        for (int i = 0; i < number.length-1; i++) {
            if (number.length == 1) {
                break;
            } else {
                if (number[i] != number[i+1]) {
                    steps++;
                }
            }
        }
        steps = steps * 2;
        return steps;
    }
    public static long ConvertArrayToLong (long[] input) {
        long output = 0;
        int multiple = input.length - 1;
        for (int i = 0; i < input.length; i++) {
            for (int j = 0; j < multiple; j++) {
                input[i] = input[i] * 10;
            }
            output = output + input[i];
            multiple--;
        }
        return output;
    }
    public static long LookSayAlgo (long input) {
        long[] inputAsArr = new long[2];
        inputAsArr = ConvertLongToArrayOfDigits(input);
        int cycles = CountSteps(inputAsArr);
        long outputLong = 0;
        long[] outputArr = new long[cycles];
        int count = 1;
        int outputCountTracker = 0;

        if (input == 1) {
            outputLong = 1;
        } else if (input == 2) {
            outputLong = 11;
        } else {
            for (int i = 0; i < inputAsArr.length; i++) {
                // do the pairs comparison
                if (i != inputAsArr.length - 1) {
                    if (inputAsArr[i] == inputAsArr[i+1]) {
                        count++;
                    } else {
                        outputArr[outputCountTracker] = count;
                        outputArr[outputCountTracker + 1] = inputAsArr[i];
                        outputCountTracker = outputCountTracker+2;
                        count = 1;
                    }
                } else { // 已經數到最後單元
                    outputArr[outputCountTracker] = count;
                    outputArr[outputCountTracker+1] = inputAsArr[i];
                    break;
                }
            }
        }
        outputLong = ConvertArrayToLong(outputArr);
        return outputLong;
    }
}
